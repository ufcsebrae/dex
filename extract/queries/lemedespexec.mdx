WITH
    MEMBER [Measures].[vl_Despesa_executado] AS 
        ([Measures].[Valor Despesa Executada])

    MEMBER [Fotografia].[Fotografia por Chave].[Intervalo] AS
        Aggregate(
            {
                [Fotografia].[Fotografia por Chave].[Fotografia].&[202401] : [Fotografia].[Fotografia por Chave].[Fotografia].&[202612]
            }
        )

SELECT
    NON EMPTY {[Measures].[vl_Despesa_executado]} ON COLUMNS,
    
    NON EMPTY (
        [Fotografia].[Fotografia].[Fotografia].Members,
        [Projetos Processos].[Nome Projeto e Processo].[Projeto e Processo].Members,
        [Ação].[Ação].[Ação].Members,
        [Unidade Organizacional de Ação].[Unidade Organizacional de Ação].[Unidade Organizacional de Ação].Members,
        [Ação].[Códgio RM].[Código RM].Members,
        [Natureza Orçamentária].[Código Natureza Orçamentária Grupo 4].[Código Natureza Orçamentária].Members,
        [Tempo].[Mês].[Mês].Members,
        [Tempo].[Ano].[Ano].Members,
        [Natureza Orçamentária].[Descrição Natureza Orçamentária Grupo 4].[Natureza Orçamentária].Members
    ) ON ROWS
FROM 
    [DWLeme]
WHERE
    (
        [Sebrae].[Sigla Sebrae].[Sebrae].&[SP],
        [Fotografia].[Fotografia por Chave].[Intervalo]
    )
